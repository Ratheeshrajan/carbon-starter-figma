import {
  DataTable,
  TableContainer,
  Table,
  TableHead,
  TableRow,
  TableHeader,
  TableBody,
  TableCell,
  Pagination,
} from '@carbon/react';
import { useState } from 'react';
import './dataTable.scss';

/**
 * Data table component showing usage analysis with pagination
 * @returns {React.ReactElement} Data table component
 */
const UsageTable = () => {
  // Sample data for the table
  const [currentPage, setCurrentPage] = useState(1);
  const [pageSize, setPageSize] = useState(10);

  // Generate sample data
  const generateData = () => {
    const data = [];
    for (let i = 0; i < 30; i++) {
      data.push({
        id: `row-${i}`,
        service: `Service ${i + 1}`,
        usage: Math.floor(Math.random() * 1000) + 100,
        status: i % 3 === 0 ? 'Active' : i % 3 === 1 ? 'Pending' : 'Inactive',
        region: i % 4 === 0 ? 'North America' : i % 4 === 1 ? 'Europe' : i % 4 === 2 ? 'Asia' : 'South America',
        lastUpdated: new Date(Date.now() - Math.floor(Math.random() * 30) * 24 * 60 * 60 * 1000).toLocaleDateString(),
      });
    }
    return data;
  };

  const data = generateData();
  
  // Calculate pagination
  const totalItems = data.length;
  const startIndex = (currentPage - 1) * pageSize;
  const endIndex = Math.min(startIndex + pageSize, totalItems);
  const currentData = data.slice(startIndex, endIndex);

  // Handle pagination change
  const handlePaginationChange = ({ page, pageSize }) => {
    setCurrentPage(page);
    setPageSize(pageSize);
  };

  // Define headers
  const headers = [
    { key: 'service', header: 'Service' },
    { key: 'usage', header: 'Usage' },
    { key: 'status', header: 'Status' },
    { key: 'region', header: 'Region' },
    { key: 'lastUpdated', header: 'Last Updated' },
  ];

  return (
    <div className="cs--dashboard__table">
      <TableContainer title="Usage analysis" description="Network usage data">
        <Table size="lg" aria-label="Usage analysis table">
          <TableHead>
            <TableRow>
              {headers.map((header) => (
                <TableHeader key={header.key}>{header.header}</TableHeader>
              ))}
            </TableRow>
          </TableHead>
          <TableBody>
            {currentData.map((row) => (
              <TableRow key={row.id}>
                <TableCell>{row.service}</TableCell>
                <TableCell>{row.usage}</TableCell>
                <TableCell>{row.status}</TableCell>
                <TableCell>{row.region}</TableCell>
                <TableCell>{row.lastUpdated}</TableCell>
              </TableRow>
            ))}
          </TableBody>
        </Table>
        <Pagination
          size="lg"
          backwardText="Previous page"
          forwardText="Next page"
          itemsPerPageText="Items per page:"
          onChange={handlePaginationChange}
          page={currentPage}
          pageSize={pageSize}
          pageSizes={[10, 20, 30, 40, 50]}
          totalItems={totalItems}
        />
      </TableContainer>
    </div>
  );
};

export default UsageTable;
